\section{–ú–∞—Ç–µ–º–∞—Ç–∏—á–µ—Å–∫–∞—è –º–æ–¥–µ–ª—å}

–†–∞—Å—Å–º–æ—Ç—Ä–∏–º –∑–∞–¥–∞—á—É —Å–±–ª–∏–∂–µ–Ω–∏—è-—É–∫–ª–æ–Ω–µ–Ω–∏—è –¥–≤—É—Ö –º–∞—Ç–µ—Ä–∏–∞–ª—å–Ω—ã—Ö —Ç–æ—á–µ–∫
–≤ –ø—Ä–æ—Å—Ç—Ä–∞–Ω—Å—Ç–≤–µ $\dimension$: 

\begin{equation}
    \label{eq:system_main}
    \frac{dx}{dt} = f_1(x,t,u) + f_2(x,t,v)
\end{equation}
 
\begin{itemize}
    \item $t \in [0,\theta]$, –≥–¥–µ $\theta$ - –º–æ–º–µ–Ω—Ç –æ–∫–æ–Ω—á–∞–Ω–∏—è –∏–≥—Ä—ã
    \item $x \in \dimension$ - —Ñ–∞–∑–æ–≤—ã–π –≤–µ–∫—Ç–æ—Ä
    \item $u(t) \in P \subset \dimension$ - —É–ø—Ä–∞–≤–ª–µ–Ω–∏–µ –ø–µ—Ä–≤–æ–≥–æ –∏–≥—Ä–æ–∫–∞
    \item $v(t) \in Q \subset \dimension$ - —É–ø—Ä–∞–≤–ª–µ–Ω–∏–µ –≤—Ç–æ—Ä–æ–≥–æ –∏–≥—Ä–æ–∫–∞
\end{itemize} 

–í—ã–±–æ—Ä –ø—Ä–∞–≤–æ–π —á–∞—Å—Ç–∏ $F(x,t,u,v) = f_1(x,t,u) + f_2(x,t,v)$
–≤ —É—Ä–∞–≤–Ω–µ–Ω–∏–∏ \ref{eq:system_main} –æ–±–µ—Å–ø–µ—á–∏—Ç –Ω–∞–º –≤—ã–ø–æ–ª–Ω–µ–Ω–∏–µ —É—Å–ª–æ–≤–∏—è –ê–π–∑–µ–∫—Å–∞:

\begin{equation}
    \label{eq:isaacs}
    \forall s, x \in \dimension :
    \min \limits_{u \in P} \max \limits_{v \in Q}
    (s, F(x,t,u,v)) =
    \max \limits_{v \in Q} \min \limits_{u \in P}
    (s, F(x,t,u,v)) 
\end{equation}
 
–í —Ç–∞–∫–æ–º —Å–ª—É—á–∞–µ –≤ –º–∞–ª–µ–Ω—å–∫–æ–π –∏–≥—Ä–µ –±—É–¥–µ—Ç —Å—É—â–µ—Å—Ç–≤–æ–≤–∞—Ç—å —Å–µ–¥–ª–æ–≤–∞—è —Ç–æ—á–∫–∞,
–ø–æ —ç—Ç–æ–º—É –¥–æ—Å—Ç–∞—Ç–æ—á–Ω–æ —Ä–∞—Å—Å–º–∞—Ç—Ä–∏–≤–∞—Ç—å –¥–∏—Ñ—Ñ–µ—Ä–µ–Ω—Ü–∏–∞–ª—å–Ω—É—é –∏–≥—Ä—É
–≤ –∫–ª–∞—Å—Å–µ –ø–æ–∑–∏—Ü–∏–æ–Ω–Ω—ã—Ö —Å—Ç—Ä–∞—Ç–µ–≥–∏–π.
–í —ç—Ç–æ–º —Å–ª—É—á–∞–µ —Å—Ç—Ä–∞—Ç–µ–≥–∏–∏ –ø–µ—Ä–≤–æ–≥–æ –∏–≥—Ä–æ–∫–∞ –∑–∞–≤–∏—Å—è—Ç
—Ç–æ–ª—å–∫–æ –æ—Ç —Ä–µ–∞–ª–∏–∑–æ–≤–∞–≤—à–µ–π—Å—è –ø–æ–∑–∏—Ü–∏–∏.

–ü—Ä–∏ –ø–æ–º–æ—â–∏ —Ñ—É–Ω–¥–∞–º–µ–Ω—Ç–∞–ª—å–Ω–æ–π –º–∞—Ç—Ä–∏—Ü—ã –ö–æ—à–∏,
–º—ã –º–æ–∂–µ–º —Å–¥–µ–ª–∞—Ç—å –∑–∞–º–µ–Ω—É –ø–µ—Ä–µ–º–µ–Ω–Ω—ã—Ö
–∏ –ø–µ—Ä–µ–π—Ç–∏ –∫ —ç–∫–≤–∏–≤–∞–ª–µ–Ω—Ç–Ω–æ–π –¥–∏—Ñ—Ñ–µ—Ä–µ–Ω—Ü–∏–∞–ª—å–Ω–æ–π –∏–≥—Ä–µ
–±–µ–∑ —Ñ–∞–∑–æ–≤–æ–π –ø–µ—Ä–µ–º–µ–Ω–Ω–æ–π –≤ –ø—Ä–∞–≤–æ–π —á–∞—Å—Ç–∏.
–í –∏—Ç–æ–≥–µ –ø–æ–ª—É—á–∏–º —Å–∏—Å—Ç–µ–º—É:

\begin{equation}
    \label{eq:system_plain}
    \frac{dy}{dt} = B(t)u + C(t)v
\end{equation}

–î–ª—è –ø—Ä–æ—Å—Ç–æ—Ç—ã –±—É–¥–µ–º —Å—á–∏—Ç–∞—Ç—å ùêµ(ùë°) –∏ ùê∂(ùë°) –º–∞—Ç—Ä–∏—Ü–∞–º–∏
—Ä–∞–∑–º–µ—Ä–Ω–æ—Å—Ç–∏ —Å–æ–æ—Ç–≤–µ—Ç—Å—Ç–≤—É—é—â–µ–π –Ω–∞—à–µ–π —Å–∏—Å—Ç–µ–º–µ, —Ç.–µ. $2\times2$

\section{–ú–∞–∫—Å–∏–º–∞–ª—å–Ω—ã–π —Å—Ç–∞–±–∏–ª—å–Ω—ã–π –º–æ—Å—Ç}

–ü—Ä–∏ –≤—ã–ø–æ–ª–Ω–µ–Ω–∏–∏ —É—Å–ª–æ–≤–∏—è \ref{eq:isaacs}
–≥–∞—Ä–∞–Ω—Ç–∏—Ä–æ–≤–∞–Ω–æ —Å—É—â–µ—Å—Ç–≤—É—é—Ç –æ–ø—Ç–∏–º–∞–ª—å–Ω—ã–µ —Å—Ç—Ä–∞—Ç–µ–≥–∏–∏
$u^0$ –∏ $v^0$: 

\begin{equation}
    \exists u^0(t,x,\epsilon), v^0(t,x,\epsilon):
    [t_0, \theta] \times \dimension \times \mathbb{R}^+
    \rightarrow \dimension
\end{equation}

–ü–æ —Ç–µ–æ—Ä–µ–º–µ –æ–± –∞–ª—Ç–µ—Ä–Ω–∞—Ç–∏–≤–µ —Å—É—â–µ—Å—Ç–≤—É–µ—Ç
–º–∞–∫—Å–∏–º–∞–ª—å–Ω—ã–π —Å—Ç–∞–±–∏–ª—å–Ω—ã–π –º–æ—Å—Ç W:

\begin{equation}
    W \subset [t_0,\theta] \times \dimension,
\end{equation}

\begin{itemize}
    \item –ø—Ä–∏–º–µ–Ω–µ–Ω–∏–µ –ø–µ—Ä–≤—ã–º –∏–≥—Ä–æ–∫–æ–º —É–ø—Ä–∞–≤–ª–µ–Ω–∏—è $u^0(t,x,\epsilon)$
    –ø—Ä–∏ $x_0 \in W(t_0)$ –≥–∞—Ä–∞–Ω—Ç–∏—Ä—É–µ—Ç $x(\theta) \in M + O(\epsilon)$
    \item –ø—Ä–∏–º–µ–Ω–µ–Ω–∏–µ –ø–µ—Ä–≤—ã–º –∏–≥—Ä–æ–∫–æ–º —É–ø—Ä–∞–≤–ª–µ–Ω–∏—è $v^0(t,x,\epsilon)$
    –ø—Ä–∏ $x_0 \notin W(t_0)$ –≥–∞—Ä–∞–Ω—Ç–∏—Ä—É–µ—Ç $x(\theta) \notin M \ominus O(\epsilon)$
\end{itemize}

–î–ª—è –ø–æ–ª—É—á–µ–Ω–∏—è –∞–ø—Ä–æ–∫—Å–∏–º–∞—Ü–∏–∏ —Å—Ç–∞–±–∏–ª—å–Ω–æ–≥–æ –º–æ—Å—Ç–∞
–¥–ª—è —Å–∏—Å—Ç–µ–º—ã \ref{eq:system_plain} –≤–æ–∑—å–º–µ–º —Ä–∞–∑–±–∏–µ–Ω–∏–µ:
\begin{equation}
    \{t_i\}, t_n = \theta
\end{equation}

–£–ø—Ä–∞–≤–ª–µ–Ω–∏—è –∏–≥—Ä–æ–∫–æ–≤ –ø–æ—Ä–æ–∂–¥–∞—é—Ç—Å—è –∫–∞–∫ –∫—É—Å–æ—á–Ω–æ-–ø–æ—Å—Ç–æ—è–Ω–Ω—ã–µ —Ñ—É–Ω–∫—Ü–∏–∏,
—Ç–µ—Ä–ø—è—â–∏–µ —Ä–∞–∑—Ä—ã–≤—ã –≤ $ùë°_i$.
–ö–∞–∂–¥–æ–µ –ø–æ—Å–ª–µ–¥—É–µ—â–µ —Å–µ—á–µ–Ω–∏–µ –º–æ—Å—Ç–∞ –≤—ã—á–∏—Å–ª—è–µ—Ç—Å—è –ø–æ —Å–ª–µ–¥—É—é—â–µ–π —Ñ–æ—Ä–º—É–ª–µ:

        \left(W_i+\int_{t_i}^{t_{i-1}} B(t) dt \cdot P\right)

    \end{aligned}
